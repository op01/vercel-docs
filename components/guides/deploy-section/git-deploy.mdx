import { IconCard } from '~/components/card'
import { List } from '~/components/list'
import BitbucketIcon from '~/components/icons/bitbucket'
import GitLabIcon from '~/components/icons/gitlab'
import GitHubIcon from '~/components/icons/github'
import ProductName from '~/components/name/product-name'
import EnvSection from './env-section'

export const Author = ({ name }) => <span>{name}</span>
export const Type = ({ type }) => <span>{type || 'project'}</span>

To deploy your <Author name={props.name} /> <Type type={props.type} /> with a [<ProductName /> for Git Integration](/docs/v2/git-integrations), make sure it has been pushed to a Git repository.

<EnvSection env={props.env} />

Import the project into <ProductName /> using your Git Integration of choice:

<List columnsDesktop={3} columnsTablet={3} columnsMobile={1}>
  <IconCard
    label="GitHub"
    icon={<GitHubIcon />}
    href={`/import/git${props.env ? '?env=' + props.env.join(',') : ''}`}
    plus
  />
  <IconCard
    label="GitLab"
    icon={<GitLabIcon />}
    href={`/import/git${props.env ? '?env=' + props.env.join(',') : ''}`}
    plus
  />
  <IconCard
    label="Bitbucket"
    icon={<BitbucketIcon />}
    href={`/import/git${props.env ? '?env=' + props.env.join(',') : ''}`}
    plus
  />
</List>

After your project has been imported, all subsequent pushes to branches will generate [Preview Deployments](/docs/v2/platform/deployments#preview), and all changes made to the [Production Branch](/docs/v2/git-integrations#production-branch) (commonly "main") will result in a [Production Deployment](/docs/v2/platform/deployments#production).
